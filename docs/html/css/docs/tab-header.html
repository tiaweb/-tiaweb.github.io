<script>
function openCity(cityName,elmnt,color) {
  var i, tiawebcontent, tiaweblinks;
  tiawebcontent = document.getElementsByClassName("tiawebcontent");
  for (i = 0; i < tiawebcontent.length; i++) {
    tiawebcontent[i].style.display = "none";
  }
  tiaweblinks = document.getElementsByClassName("tiaweblink");
  for (i = 0; i < tiaweblinks.length; i++) {
    tiaweblinks[i].style.backgroundColor = "";
  }
  document.getElementById(cityName).style.display = "block";
  elmnt.style.backgroundColor = color;

}
// Get the element with id="defaultOpen" and click on it
document.getElementById("defaultOpen").click();
</script>
