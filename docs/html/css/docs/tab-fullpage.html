<script>
function openPage(pageName,elmnt,color) {
  var i, tiawebtab3content, tiawebtab3links;
  tiawebtab3content = document.getElementsByClassName("tiawebtab3content");
  for (i = 0; i < tiawebtab3content.length; i++) {
    tiawebtab3content[i].style.display = "none";
  }
  tiawebtab3links = document.getElementsByClassName("tiawebtab3link");
  for (i = 0; i < tiawebtab3links.length; i++) {
    tiawebtab3links[i].style.backgroundColor = "";
  }
  document.getElementById(pageName).style.display = "block";
  elmnt.style.backgroundColor = color;
}

// Get the element with id="defaultOpen" and click on it
document.getElementById("defaultOpen").click();
</script>
