<script>
function openCity(evt, cityName) {
  var i, tiawebtabcontent, tiawebtablinks;
  tiawebtabcontent = document.getElementsByClassName("tiawebtabcontent");
  for (i = 0; i < tiawebtabcontent.length; i++) {
    tiawebtabcontent[i].style.display = "none";
  }
  tiawebtablinks = document.getElementsByClassName("tiawebtablinks");
  for (i = 0; i < tiawebtablinks.length; i++) {
    tiawebtablinks[i].className = tiawebtablinks[i].className.replace(" active", "");
  }
  document.getElementById(cityName).style.display = "block";
  evt.currentTarget.className += " active";
}

// Get the element with id="defaultOpen" and click on it
document.getElementById("defaultOpen").click();
</script>
