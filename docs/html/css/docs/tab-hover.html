<script>
function openCity(evt, cityName) {
  var i, tiawebtab4content, tiawebtab4links;
  tiawebtab4content = document.getElementsByClassName("tiawebtab4content");
  for (i = 0; i < tiawebtab4content.length; i++) {
    tiawebtab4content[i].style.display = "none";
  }
  tiawebtab4links = document.getElementsByClassName("tiawebtab4links");
  for (i = 0; i < tiawebtab4links.length; i++) {
    tiawebtab4links[i].className = tiawebtab4links[i].className.replace(" active", "");
  }
  document.getElementById(cityName).style.display = "block";
  evt.currentTarget.className += " active";
}
</script>
