<script>
function openCity(evt, cityName) {
  var i, tiawebtab1content, tiawebtab1links;
  tiawebtab1content = document.getElementsByClassName("tiawebtab1content");
  for (i = 0; i < tiawebtab1content.length; i++) {
    tiawebtab1content[i].style.display = "none";
  }
  tiawebtab1links = document.getElementsByClassName("tiawebtab1links");
  for (i = 0; i < tiawebtab1links.length; i++) {
    tiawebtab1links[i].className = tiawebtab1links[i].className.replace(" active", "");
  }
  document.getElementById(cityName).style.display = "block";
  evt.currentTarget.className += " active";
}

// Get the element with id="defaultOpen" and click on it
document.getElementById("defaultOpen").click();
</script>
